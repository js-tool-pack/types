Type alias PublicOnly<T>

PublicOnly<T>: Pick<T, keyof T>

pick所有public属性组成新的type

Example

class Foo {
  public a = '';
  protected b = 2;
  private c = false;

  constructor() {
    console.log(this.c);
  }
}
type T = PublicOnly<Foo>; // {a: string}
